        /// <summary>
        /// PC搜索页同步筛选条件
        /// </summary>
        /// <description>allParams表示地址中的参数-0-1-0-0-0-1-1或者s1g2c3r4t5m210611d8，
        其中以-开头的参数需要写死-开头，顺序必须按照指定顺序使用。
        第二种参数s1g2c3r4t5m210611d8按需传递，需要什么参数传入什么参数，顺序可变换</description>
        /// <author>FreshMan</author>
        /// <createtime>2016-11-09</createtime>
        /// <returns>搜索接口返回原始数据</returns>
        public ApiResultEntity<AbsResponseCruiseSearchModel> GetSearchConditionInfo()
        {
            //获取参数
            var strParam = WebRequestHelper.GetStringFromParameters( "allParams");
            var platform = WebRequestHelper.GetIntFromParameters( "platform");
            var dest = HttpUtility.UrlDecode(WebRequestHelper.GetStringFromParameters( "dest"));
            var lcCityId = WebRequestHelper.GetIntFromParameters( "lc");
            var ipAddress = WebRequestHelper.GetStringFromParameters( "ipAddress");
            var platformEnum = EnumHelper.GetEnumByValue<SiteTypeEnum>(platform);
            //选择使用平台上下文服务
            ICruiseSearchContextServer platformContext = CruiseSearchContextFactory.GetSearchContextFatory(platformEnum);
            //确定使用何种接口获取数据
            platformContext.SearchDataServer = CruiseSearchDalServer.GetSearchDataServer(SearchInterfaceEnum.Dsf);
            //确定使用何种装饰
            platformContext.CruiseSearchDecorateServer =  new  CruiseSearchDecorateSearchConditionServer();
            //参数解析
            var searchParam = CruiseSearchParamServerBll.AnalysisSearchParams(strParam);
            if (searchParam !=  null  && searchParam.LcCityId < 1) searchParam.LcCityId = lcCityId;
            //调取自定义请求参数类型转换方法
            var searchModel =  new  TurenSearchParamsToAbsSearchModelServer()
            .DefaultTurnSearchParamsToAbsSearchModel(searchParam, dest, ipAddress, 1);
            var searchLineInfoList =
             new CruiseSearchContextLineServerBll(platformContext).GetCruiseSearchContextInfo(searchModel,  null);
            return HandleSuccessResult(searchLineInfoList);
        }
